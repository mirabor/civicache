```cpp
// Standalone W-TinyLFU + CountMinSketch test binary.
// Pure C++17 assert() per CONTEXT.md D-06 — no Catch2/googletest.
// Run: make test (builds build/test/test_wtinylfu and runs it).
//
// Coverage (CONTEXT.md D-05):
//   T1. CMS basics: record a key N times; estimate(key) in [N, N+3] (bounded tolerance).
//   T2. CMS aging: force_age() halves all counters and resets sample_count to 0.
//   T3. Hot-survives-scan (WTLFU-04 literal): hot key accessed 20x, then 1000 unique
//       scan accesses, hot key still in cache (confirmed via access() returning true).
//   T4. Determinism: two back-to-back runs with the same input sequence produce
//       identical access()-hit-sequences (observable cache behavior is deterministic).

#include <cassert>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <string>
#include <vector>
#include "count_min_sketch.h"
#include "cache.h"  // brings in CachePolicy + WTinyLFUCache (via cache.h's subordinate include of wtinylfu.h)

static int failures = 0;
#define TEST_ASSERT(expr, test_name) do { \
    if (!(expr)) { \
        std::fprintf(stderr, "FAIL: %s — assertion \"%s\" failed at %s:%d\n", \
                     test_name, #expr, __FILE__, __LINE__); \
        ++failures; \
    } \
} while (0)

// T1: CMS basics — record N times, estimate within [N, min(N+3, 15)].
static void test_cms_basics() {
    const char* tn = "cms_basics";
    CountMinSketch cms(1024);
    assert(cms.width() >= 1024 && (cms.width() & (cms.width() - 1)) == 0);  // nextpow2

    const std::string key = "hot_key";
    for (int i = 0; i < 10; ++i) cms.record(key);
    uint32_t est = cms.estimate(key);
    // Standard CMS: est is an OVER-estimate (min across rows); true count is 10.
    // Tolerance +3 per D-05 first bullet (Claude's Discretion says this is a starting
    // guess — tune only if test flakes in CI). Saturate at COUNTER_MAX=15.
    TEST_ASSERT(est >= 10 && est <= 15, tn);

    // A never-recorded key should estimate 0 (with probability near 1 — no collision
    // with "hot_key" on ALL four rows simultaneously is extremely unlikely at width>=1024).
    uint32_t zero_est = cms.estimate("never_recorded_key_xyz");
    TEST_ASSERT(zero_est == 0, tn);

    if (failures == 0) std::fprintf(stderr, "PASS: %s\n", tn);
}

// T2: CMS aging — force_age() halves all counters; sample_count resets.
static void test_cms_aging() {
    const char* tn = "cms_aging";
    CountMinSketch cms(256);

    // Record "A" twice — estimate should be 2 (or saturated, but 2 is comfortably
    // below saturation for a fresh sketch).
    cms.record("A");
    cms.record("A");
    uint32_t before = cms.estimate("A");
    TEST_ASSERT(before >= 2, tn);

    cms.force_age();  // immediately halve all counters; sample_count -> 0.

    uint32_t after = cms.estimate("A");
    TEST_ASSERT(after == before / 2, tn);
    TEST_ASSERT(cms.sample_count() == 0, tn);

    // Record a burst large enough to approach natural aging trigger. Use a small
    // width (256) and record width*DEPTH*10 = 256*4*10 = 10240 times across many keys
    // to drive the natural aging code path at least once. After the burst,
    // sample_count must have cycled through >= 1 halving.
    cms.reset();
    uint64_t burst = cms.sample_size() + 10;
    for (uint64_t i = 0; i < burst; ++i) {
        cms.record("key_" + std::to_string(i % 5000));
    }
    // After sample_size records, aging fires automatically and sample_count resets to 0.
    // Current sample_count is therefore (burst - sample_size) = 10 (exactly).
    TEST_ASSERT(cms.sample_count() < cms.sample_size(), tn);

    if (failures == 0) std::fprintf(stderr, "PASS: %s\n", tn);
}

// T3: Hot-object-survives-scan (WTLFU-04 literal).
// Insert hot key, access 20x, then access 1000 UNIQUE scan keys sequentially.
// Confirm hot key is still in cache by re-accessing it and observing a hit.
static void test_hot_survives_scan() {
    const char* tn = "hot_survives_scan";

    // Size cache so that 1000 unique objects CANNOT all fit but the hot key CAN.
    // Each object = 100 bytes; 1000 scan objects = 100,000 bytes total.
    // Cache = 20,000 bytes = room for ~200 objects at most. Scan must overflow.
    // n_objects_hint matches the ~200 expected residents.
    const uint64_t cap_bytes = 20000;
    const uint64_t obj_size = 100;
    const uint64_t n_obj_hint = 200;
    WTinyLFUCache cache(cap_bytes, n_obj_hint);

    const std::string hot = "HOT";
    // 20 accesses to establish hot key's CMS frequency.
    for (int i = 0; i < 20; ++i) (void)cache.access(hot, obj_size);

    // Sequential scan of 1000 unique keys. Each miss; none should bump hot
    // out of the cache because after main admission starts, hot's CMS count (>=20
    // across 20 accesses) dominates any scan key's count (1 each).
    for (int i = 0; i < 1000; ++i) {
        std::string scan_key = "scan_" + std::to_string(i);
        (void)cache.access(scan_key, obj_size);
    }

    // Re-access hot. Should return true (hit) because hot's CMS frequency
    // dominates any scan key's, so W-TinyLFU's admission would have kept hot
    // resident across the scan.
    bool hot_hit = cache.access(hot, obj_size);
    TEST_ASSERT(hot_hit, tn);

    if (hot_hit) std::fprintf(stderr, "PASS: %s\n", tn);
}

// T4: Determinism — two back-to-back identical runs produce identical access-hit sequences.
// Observable cache behavior must be deterministic given the same seed (FNV-1a seeds are
// hard-coded in hash_util.h; CountMinSketch uses them; no wall-clock or RNG inputs).
static void test_determinism() {
    const char* tn = "determinism";

    auto run_once = [](std::vector<bool>& hits) {
        WTinyLFUCache cache(20000, 200);
        const uint64_t obj_size = 100;
        for (int i = 0; i < 500; ++i) {
            // Mix a hot key every 5th access with unique scan keys otherwise.
            std::string key = (i % 5 == 0) ? "HOT" : "scan_" + std::to_string(i);
            hits.push_back(cache.access(key, obj_size));
        }
    };

    std::vector<bool> run1, run2;
    run_once(run1);
    run_once(run2);

    TEST_ASSERT(run1.size() == run2.size(), tn);
    TEST_ASSERT(run1 == run2, tn);

    if (run1 == run2) std::fprintf(stderr, "PASS: %s\n", tn);
}

int main() {
    std::fprintf(stderr, "=== W-TinyLFU + CountMinSketch test suite ===\n");
    test_cms_basics();
    test_cms_aging();
    test_hot_survives_scan();
    test_determinism();
    if (failures > 0) {
        std::fprintf(stderr, "\n%d test(s) FAILED.\n", failures);
        return 1;
    }
    std::fprintf(stderr, "\nAll tests PASSED.\n");
    return 0;
}
```

Implementation notes:

1. The test binary `#include "cache.h"` — which (after 02-03 Task 2) transitively includes `wtinylfu.h` and `count_min_sketch.h`. This matches the `make_policy` consumer pattern.

2. The `TEST_ASSERT` macro prints a `FAIL:` one-liner and increments `failures`. We accumulate across all tests rather than aborting on first failure so a `make test` run surfaces every broken invariant in one run. Exit code is `(failures > 0) ? 1 : 0` per D-06.

3. T1 tolerance of `[10, 15]` respects standard CMS overcounting (never undercounts) and 4-bit saturation. If the 02-02 §2 lock chose CONSERVATIVE update, the estimate will typically be `== N` exactly until collisions push it up; STANDARD update allows more spread. Both are accommodated.

4. T2's second phase drives the natural aging trigger by recording `sample_size + 10` times. After natural aging fires, `sample_count` is `< sample_size`. If the 02-02 implementation uses a different reset formula (e.g., `sample_count -= sample_size`), this test's third assertion still holds.

5. T3 is the literal WTLFU-04 check. Sizing: cache=20000 bytes, obj_size=100 bytes → ~200-object capacity. Scan is 1000 unique, so scan alone overflows by 5×. Hot key accessed 20× before the scan starts; its CMS count of ~20 dominates any scan key's count of 1 under D-08e strict-greater admission.

6. T4 relies on the FNV-1a seeds being hard-coded in `hash_util.h` and no wall-clock inputs in CMS or W-TinyLFU. If a future refactor introduces non-determinism (e.g., `std::random_device` or `steady_clock::now()` inside the policy), this test will catch it.

7. The test file is self-contained — it does NOT load `traces/congress_trace.csv`. This is intentional per CONTEXT.md §specifics: "Test binary has no trace dependency … Grader can `make test` in a clean checkout."
