```python
# Pattern: for each seed, for each workload, run ./cache_sim and rename output
SEEDS = [42, 7, 13, 23, 31]   # D-05 seed set; 42 is existing default

WORKLOADS = {
    "congress": "traces/congress_trace.csv",
    "court":    "traces/court_trace.csv",
}

def run_cell(cache_sim, workload, trace, seed, output_base, dry_run):
    """Run one (workload, seed) cell and rename produced CSVs to per-seed names."""
    out_dir = os.path.join(output_base, workload)
    os.makedirs(out_dir, exist_ok=True)
    cmd = [
        cache_sim,
        "--trace", trace,
        "--replay-zipf",
        "--alpha-sweep",
        "--seed", str(seed),
        "--output-dir", out_dir,
    ]
    if dry_run:
        print("  DRY-RUN:", " ".join(cmd))
        return
    print(f"  Running: {' '.join(cmd)}")
    result = subprocess.run(cmd, check=False)
    if result.returncode != 0:
        print(f"  ERROR: cache_sim exited {result.returncode} for {workload} seed={seed}",
              file=sys.stderr)
        sys.exit(1)
    # Rename per-seed: mrc.csv -> mrc_seed42.csv, alpha_sensitivity.csv -> alpha_sensitivity_seed42.csv
    for stem in ("mrc", "alpha_sensitivity"):
        src = os.path.join(out_dir, f"{stem}.csv")
        dst = os.path.join(out_dir, f"{stem}_seed{seed}.csv")
        if os.path.exists(src):
            os.rename(src, dst)
            print(f"  Renamed {src} -> {dst}")
```

**Welch's t-test pattern** (D-02):
```python
from scipy import stats

def welch_vs_lru(group_df, policy, lru_vals):
    """Return (p_value, significant) for policy vs LRU miss_ratio across 5 seeds."""
    policy_vals = group_df[group_df["policy"] == policy]["miss_ratio"].values
    if len(policy_vals) < 2 or len(lru_vals) < 2:
        return (float("nan"), False)
    _, p = stats.ttest_ind(policy_vals, lru_vals, equal_var=False)
    return (p, p < P_SIG)
```

```python
def check_sc1_outputs(compare_dir):

    fails = []

    for ext in ("workload_characterization.md", "workload_characterization.json",
                "winner_per_regime.md", "winner_per_regime.json"):
        p = os.path.join(compare_dir, ext)

            fails.append(f"Missing output: {p}")
    return fails

def check_sc2_seeds(compare_dir):
    """SC-2: all 5 seed files present for both workloads and both stems."""
    fails = []
    for wl in WORKLOADS:
        for stem in STEMS:
            for seed in SEEDS:
```
